Assignar i mostrar

Assignació de 0 a n-1 a un vector


Assignar a un vector d'n elements els successius nombres naturals (0, 1, 2, 3, ...). 
void assignar_vector_enter(Vector_enter vec, unsigned n) {
    //Pre: 0<=n<=N_MAX
    //Post: vec[0..n-1] conté successivament els naturals des de 0 a n-1
    for (unsigned i=0; i<n; i++) vec[i]=i;
}
void escriure_vector_enter(const Vector_enter vec, unsigned n) {
    //Pre: 0<=n<=N_MAX
    //Post: s'han mostrat el n valors de vec[0..n-1]
    for (unsigned i=0; i<n; i++) cout<<vec[i]<<" ";
}

int main() {   
    Vector_enter vec;
    unsigned n;
    cout<<"Nombre elements del vector: ";
    cin>>n; 
    assignar_vector_enter(vec,n);
    
    cout<<"Vector obtingut: "<<endl;
    escriure_vector_enter(vec,n);
    cout<<endl;
    return 0;
}

Assignació de 0 a n*m-1 a una matriu


Assignar a una matriu n*m els successius nombres naturals per files. 
void assigna_matriu_enter(Matriu_enter mat, unsigned n, unsigned m) {
    //Pre: 0<=n<=N_FIL_MAX, 0<=m<=N_COL_MAX
    //Post: mat[0..n-1,0..m-1] conté successivament els naturals des de 0 a n*m-1
    unsigned comptador=0;
    for (unsigned i=0; i<n; i++)
        for (unsigned j=0; j<m; j++) {
            mat[i][j]=comptador;
            comptador++;
        }
}

void escriure_matriu_enter(const Matriu_enter mat, unsigned n, unsigned m) {
    //Pre: 0<=n<=N_FIL_MAX, 0<=m<=N_COL_MAX
    //Post: s'han mostrat el n*m valors de mat[0..n-1,0..m-1]
    for (unsigned i=0; i<n; i++) {
        for (unsigned j=0; j<m; j++)
            cout<<setw(5)<<mat[i][j];
        cout<<endl;
    }
}
int main() {
    Matriu_enter mat;
    unsigned n,m;
    cout<<"Nombre files i columnes de la matriu: ";
    cin>>n>>m;
    
    assigna_matriu_enter(mat,n,m);
    cout<<endl;
    
    cout<<"Matriu obtinguda: "<<endl;
    escriure_matriu_enter(mat,n,m);
    cout<<endl;
    
    return 0;
}

Llegir fins a marca de fi

Omplir un vector des de teclat i amb marca de fi 0


Fer un subprograma que permeti omplir un vector des de teclat i amb marca de fi -1. 
void llegir_vector_natural_amb_fi(Vector_natural vec, unsigned &n) {
    //Pre: cert
    //Post: 0<=n<=N_MAX i vec[0..n-1] conté n valors entrats
    n=0;
    unsigned element;
    cin>>element; // marca de fi 0
    while (n<N_MAX and element>0) {
        vec[n]=element;
        n++;
        cin>>element; // marca de fi 0
    }
}
int main() {  
    Vector_natural vec;
    unsigned n;    
    cout<<"Entra naturals (fi amb 0): ";
    llegir_vector_natural_amb_fi(vec,n);
    cout<<endl;
    
    cout<<"Vector entrat: "<<endl;
    escriure_vector_natural(vec,n);
    cout<<endl;
    
    return 0;
}

Omplir un vector de tuples amb marca de fi 0


Fer un subprograma que permeti omplir un vector de tuples Producte des de teclat i amb marca de fi codi=-1. 
const unsigned N_MAX=100;

struct Producte {
    unsigned codi;
    double preu;
};
typedef Producte Vector_producte[N_MAX];

void llegir_vector_producte_amb_fi(Vector_producte vec, unsigned &n) {
    //Pre: cert
    //Post: 0<=n<=N_MAX i vec[0..n-1] conté n productes entrats
    n=0;
    unsigned codi;
    cin>>codi;
    while (n<N_MAX and codi>0) {
        vec[n].codi=codi;
        cin>>vec[n].preu;
        n++;
        cin>>codi;
    }
}

void escriure_vector_producte(const Vector_producte vec, unsigned n) {
    //Pre: 0<=n<=N_MAX
    //Post: s'han mostrat els n productes de vec[0..n-1]
    for(unsigned i=0; i<n; i++) cout<<vec[i].codi<<" - "<<vec[i].preu<<endl;
}
int main() {
    
    Vector_producte vec;
    unsigned n;
    
    cout<<"Entra productes amb fi codi=0: ";
    llegir_vector_producte_amb_fi(vec,n);
    cout<<endl;

    cout<<"Vector entrat: "<<endl;
    escriure_vector_producte(vec,n);
    cout<<endl;
    
    return 0;
}
