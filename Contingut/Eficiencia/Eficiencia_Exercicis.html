                             void intercanvi(int& x, int& y) {
                                int aux=x; x=y; y=aux;                                     
                            }

                            int pos_minim_des_de(const Vector_enter vec, unsigned m, unsigned pos) {
                                //Pre: 1<m=N_MAX i 0<=pos<m
                                //Post: retorna la posició del mínim de vec[pos..m-1]
                                int pos_min=pos;                                   
                                for (unsigned i=pos+1; i<m; i++) 
                                    if(vec[i]<vec[pos_min]) pos_min=i;   

                                return pos_min;                                
                            }


                        //PROGRAMA
                            void ordenacio_seleccio_directa(Vector_enter vec, unsigned m) {

                                //Pre: 0<m<=N_MAX i vec=VEC
                                //Post: vec[0..m-1] conté una permutació ordenada dels valors de VEC[0..m-1]
                                for (unsigned i=0; i<m-1; i++) {
                                    unsigned pos_min= pos_minim_des_de(vec,m,i);  
                                    intercanvi(vec[i],vec[pos_min]);	
                                }
                            }
    • Complexitat en funció de els dades d’entrada: O(ordenació_selecció_directe) = O(m2)
    • Complexitat en funció de la grandària definida com (fer canvi de variable)
      1. el n. d’elem. del vector n=m: O(ordenació_selecció_directe) = O(n2) amb n=m

                        bool diagonal_positiva(Matriu_enter matriu){
                        //Pre: matriu.n<=N_MAX_FILES i matriu.m <= N_MAX_COL  i matriu.n=matriu.m
                        //Post: retorna cert matriu.mat[0..matriu.n-1,0..matriu.m-1] té la diagonal >=0
                        
                            bool positiva=true;                       
                            unsigned i=0;                              
                            while(i<matriu.n and positiva){            
                                if(matriu.mat[i][i]<0) positiva=false;   
                                else i++;                                                                                              
                            }
                            return positiva;                         

                        }
    • En funció de les dades d'entrada: O(diagonal_positiva) = O(matriu.n)
    • Complexitat en funció de la grandària que es defineix com (fer canvi de variable).
      1. el número d’elements de la matriu n=matriu.n*matriu.m=matriu.n2:
      O(diagonal_positiva) = O(sqrt(n)) amb n=matriu.n2
